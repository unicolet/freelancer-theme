Configuration file reference

The following is the default configuration file that ships with Middleware Insight. 

#----------------------------------------------
#
# Configuration for the Mirth Connect database
# 
# The values in this section should be the same
# set in the Mirth Connect configuration file.
#
#----------------------------------------------
#
# Example for postgres
jdbc.driver=org.postgresql.Driver
jdbc.url=jdbc:postgresql://127.0.0.1:5432/mirth322

# Example for oracle
#jdbc.driver=oracle.jdbc.driver.OracleDriver
#jdbc.url=jdbc:oracle:thin:@(DESCRIPTION=(ADDRESS=(PROTOCOL=TCP)(HOST=<host>)(PORT=<port>))(CONNECT_DATA=(SID=ORCL)))

# Example for mysql
#jdbc.driver=com.mysql.jdbc.Driver
#jdbc.url=jdbc:mysql://<server>:<port>/<database>

# Example for sqlserver
#jdbc.driver=net.sourceforge.jtds.jdbc.Driver
#jdbc.url=jdbc:jtds:sqlserver://<server>:<port>/<database>

# mirth database username and password
jdbc.user=mirth
jdbc.password=mirth

#----------------------------------------------
#
# Elasticsearch configuration
#
#----------------------------------------------
#
#  name of the Eleasticsearch cluster
es.clustername=mirth
# ip of the elasticsearch cluster
es.ip=127.0.0.1
# the timestamped prefix for the Elasticsearch index name. Middleware Insight will generate
# one index per day with following pattern: es.index.prefix-yyyy.MM.dd
es.index.prefix=mirth

# Middleware Insight will not fetch messages created before this date. Useful if your database
# contains a lot of old messages and you don't care about them or do not want to
# overload the database and Elasticsearch in the first run
# The installation script will set this parameter to the installation date.
sql.date.limit=01/09/2016

# A javascript function that can be used to remap the status of messages while
# being imported from Mirth into Elasticsearch.
# There is an example script in: /usr/share/doc/insight
status.decoder.script=classpath:status_decoder.js

# Middleware insight exports daily stats at this port in json format:
# to view them: curl http://127.0.0.1:8888/stats
stats.port=8888

# internal settings, do not change them
version=1
target=mirth
